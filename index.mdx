```bash npm
npm install stake-engine
```

```typescript
import { RGSClient } from 'stake-engine';

const client = RGSClient({ url: window.location.href });

// Authenticate the player session
const { balance, config, jurisdictionFlags } = await client.Authenticate();

// Place a bet using a valid bet level
const result = await client.Play({
  amount: config.betLevels[0],
  mode: 'base',
});

// Complete the round when game logic resolves
await client.EndRound();
```
